#include <Stepper.h>
double spr = 2048;
Stepper ms(spr,8,10,9,11);
const int buttonPin1 = 1; 
const int buttonPin2 = 2; 
const int buttonPin3 = 3; 
int position = 1;
setup(){
ms.setSpeed(10);
}
loop(){
if (digitalRead(buttonPin1) == HIGH && digitalRead(buttonPin2) == LOW && digitalRead(buttonPin3) == LOW){
if (position == 1){
position = 1;
}
else if (position == 2){
ms.step(-400);
delay(1000);
position = 1;
}

else if (position == 3){
ms.step(-800);
delay(1000);
position = 1;
}

}



else if (digitalRead(buttonPin2) == HIGH && digitalRead(buttonPin1) == LOW && digitalRead(buttonPin3) == LOW){
if (position == 1){
ms.step(400);
delay(1000);
position = 2;
}
else if (position == 2){
position = 2;
}

else if (position == 3){
ms.step(-400);
delay(1000);
position = 2;
}

}



else if (digitalRead(buttonPin3) == HIGH && digitalRead(buttonPin1) == LOW && digitalRead(buttonPin2) == LOW){
if (position == 1){
ms.step(800);
delay(1000);
position = 3;
}
else if (position == 2){
ms.step(400);
delay(1000);
position = 3;
}

else if (position == 3){

position = 3;
}

}



}
